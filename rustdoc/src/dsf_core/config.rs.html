1use crate::backend::StackedBackendConfig;
2use colored::Colorize;
3use directories::ProjectDirs;
4use serde::{Deserialize, Serialize};
5use std::fs;
6use std::io::{self, Error, ErrorKind};
7use std::path::PathBuf;
8
9#[derive(Debug, Clone, PartialEq, Default, Serialize, Deserialize)]
10#[serde(rename_all = "lowercase")]
11pub enum InstallMode {
12    #[default]
13    User,
14    Global,
15}
16
17#[derive(Debug, Clone, Serialize, Deserialize)]
18pub struct AppConfig {
19    #[serde(skip)]
20    pub mode: InstallMode,
21    #[serde(skip)]
22    pub config_path: Option<PathBuf>,
23    pub backend: StackedBackendConfig,
24}
25
26impl AppConfig {
27    pub(crate) fn resolve_path() -> io::Result<(InstallMode, PathBuf)> {
28        // P1: XDG 标准用户目录探测
29        let proj_dirs =
30            ProjectDirs::from("io", "flyingbucket", "dataspringflow").ok_or_else(|| {
31                io::Error::new(
32                    io::ErrorKind::NotFound,
33                    "Failed to resolve user home directory".red().to_string(),
34                )
35            })?;
36
37        let user_cfg_path = proj_dirs.config_dir().join("config.yaml");
38        let global_cfg_path = PathBuf::from("/etc/dataspringflow/config.yaml");
39
40        if user_cfg_path.exists() {
41            Ok((InstallMode::User, user_cfg_path))
42        } else if global_cfg_path.exists() {
43            Ok((InstallMode::Global, global_cfg_path))
44        } else {
45            let err_msg = format!(
46                "{}\n{} Run `{}` first.",
47                "Neither user installation nor global installation detected on this server.".red(),
48                "Suggestion:".bright_blue().bold(),
49                "dsf init".green().underline()
50            );
51
52            Err(io::Error::new(io::ErrorKind::NotFound, err_msg))
53        }
54    }
55    pub fn load() -> io::Result<Self> {
56        let (mode, config_path) = Self::resolve_path()?;
57
58        let content = fs::read_to_string(&config_path).map_err(|e| {
59            Error::other(format!(
60                "Read config failed ({}): {e}",
61                config_path.display()
62            ))
63        })?;
64
65        let mut app_cfg = serde_yaml::from_str::<Self>(&content).map_err(|e| {
66            let err_msg = format!(
67                "{}\n({}):{e}",
68                "Parse config yaml failed".red(),
69                config_path.display()
70            );
71            Error::new(ErrorKind::InvalidData, err_msg)
72        })?;
73
74        app_cfg.mode = mode;
75        app_cfg.config_path = Some(config_path);
76
77        Ok(app_cfg)
78    }
79
80    pub fn get_local_global_path() -> io::Result<PathBuf> {
81        let local_global = PathBuf::from("/etc/dataspringflow/config.yaml");
82        if !local_global.exists() {
83            let err_msg = format!(
84                "{}: {}",
85                "Local global sqlite config file not found".red().bold(),
86                local_global.to_string_lossy()
87            );
88            return Err(io::Error::new(io::ErrorKind::NotFound, err_msg));
89        }
90        Ok(local_global)
91    }
92}
