5impl From<MetaData> for MetaDetailDto {
6    fn from(meta: MetaData) -> Self {
7        let id = format!("{}@{}", meta.name, meta.tag);
8
9        let busy_status = meta.busy_status.as_str().to_string();
10
11        Self {
12            id,
13            name: meta.name,
14            tag: meta.tag,
15            hash: meta.hash,
16            path: meta.path.to_string_lossy().into_owned(),
17            description_path: meta.description_path.to_string_lossy().into_owned(),
18            script_path: meta.script_path.to_string_lossy().into_owned(),
19            owner: meta.owner,
20            dependencies: meta.dependencies,
21            merkle_tree_path: meta.merkle_tree_path.to_string_lossy().into_owned(),
22            busy_status,
23        }
24    }
25}

27impl From<DatasetGraph> for WebGraphResponse {
28    fn from(value: DatasetGraph) -> Self {
29        let mut nodes = Vec::new();
30        let mut edges = Vec::new();
31
32        for (id, dataset) in &value.datasets {
33            nodes.push(WebGraphNode {
34                id: id.clone(),
35                status: dataset.detailed_status.status.to_string(),
36                owner: Some(dataset.metadata.owner.clone()),
37            });
38        }
39
40        for (node_id, deps) in &value.adj_list {
41            for dep_id in deps {
42                edges.push(WebGraphEdge {
43                    // 表示 node_id 依赖于 dep_id
44                    source: node_id.clone(),
45                    target: dep_id.clone(),
46                });
47            }
48        }
49
50        WebGraphResponse { nodes, edges }
51    }
52}
