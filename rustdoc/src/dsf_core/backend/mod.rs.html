20pub trait DatasetBackend {
21    /// Retrieves the corresponding metadata by the dataset ID.
22    fn get_metadata(&self, id: &str) -> BackendResult<MetaData>;
23
24    /// Mark MetaData status to ensure disk data and backend metadata consistency
25    fn mark_status(&self, id: &str, status: DataSetBusyStatus) -> BackendResult<()>;
26    /// Saves or updates the dataset metadata.
27    fn save_metadata(&self, metadata: &MetaData) -> BackendResult<()>;
28    /// Checks if any datasets depend on the specified `target_id`.
29    ///
30    /// Returns a list of dataset IDs that reference the target.
31    fn check_is_referenced(&self, target_id: &str) -> BackendResult<Vec<String>>;
32    /// Lists all available dataset metadata from the backend.
33    fn list_all_metadata(&self) -> BackendResult<Vec<MetaData>>;
34    /// Deletes the metadata associated with the specified dataset ID.
35    /// note: this mucntion only deletes the metadata and detach this dataset from backend regisitration,
36    /// real data on disk will be safe
37    fn delete_metadata(&self, id: &str) -> BackendResult<()>;
38}

43pub enum BackendError {
44    // 核心业务与状态错误
45    #[error("Dataset not found: {id}")]
46    DatasetNotFound { id: String },
47
48    #[error("Target global backend is either unreachable or not found in config")]
49    BackendNotFound,
50
51    #[error("Dataset already exists: {id}")]
52    AlreadyExists { id: String },
53
54    #[error("Permission denied: {message}")]
55    PermissionDenied { message: String },
56
57    #[error("Unsupported operation: {message}")]
58    Unsupported { message: String },
59
60    #[error("Invalid config: {message}")]
61    InvalidConfig { message: String },
62
63    // 存储与底层错误
64    #[error("Database pool error: {message}")]
65    PoolError { message: String },
66
67    #[error("Failed setting pragma: {message}")]
68    SetPragma { message: String },
69
70    #[error("Database execution failed: {source}")]
71    StorageError {
72        #[from]
73        source: rusqlite::Error,
74    },
75
76    #[error("Data serialization or parsing failed: {message}")]
77    SerializationError { message: String },
78
79    #[error("Internal I/O error: {source}")]
80    Io {
81        #[from]
82        source: std::io::Error,
83    },
84
85    #[error("Metadata core error: {source}")]
86    MetaData {
87        #[from]
88        source: crate::core::MetaDataError,

92impl BackendError {
93    pub fn to_io_error(self) -> io::Error {
94        match self {
95            BackendError::Io { source } => source,
96            other => {
97                let kind = match &other {
98                    BackendError::DatasetNotFound { .. } => io::ErrorKind::NotFound,
99                    BackendError::BackendNotFound => io::ErrorKind::NotFound,
100                    BackendError::AlreadyExists { .. } => io::ErrorKind::AlreadyExists,
101                    BackendError::PermissionDenied { .. } => io::ErrorKind::PermissionDenied,
102                    _ => io::ErrorKind::Other,
103                };
104                io::Error::new(kind, other)
105            }
106        }
107    }
108
109    pub fn to_metadata_error(self) -> MetaDataError {
110        match self {
111            BackendError::MetaData { source } => source,
112            other => {
113                let io_err = other.to_io_error();
114                MetaDataError::Io(io_err)
115            }
116        }
117    }
118
119    pub fn to_dag_error(self) -> DatasetGraphError {
120        match self {
121            BackendError::DatasetNotFound { id } => {
122                DatasetGraphError::DatasetNotFound { node_id: id }
123            }
124
125            BackendError::MetaData { source } => {
126                if let crate::core::MetaDataError::Io(io_err) = source {
127                    DatasetGraphError::Io(io_err)
128                } else {
129                    let io_err =
130                        std::io::Error::new(std::io::ErrorKind::InvalidData, source.to_string());
131                    DatasetGraphError::Io(io_err)
132                }
133            }
134
135            BackendError::Io { source } => DatasetGraphError::Io(source),
136            other => {
137                let io_err = other.to_io_error();
138                DatasetGraphError::Io(io_err)
139            }
140        }
141    }
142}

146pub fn capture_backtrace() {
147    struct StackFrame {
148        name: String,
149        file: String,
150        line: u32,
151    }
152
153    let mut frames = Vec::new();
154
155    backtrace::trace(|frame| {
156        backtrace::resolve_frame(frame, |symbol| {
157            if let Some(name) = symbol.name() {
158                let symbol_name = name.to_string();
159
160                let file_path = symbol
161                    .filename()
162                    .map(|p| p.to_string_lossy().into_owned())
163                    .unwrap_or_else(|| "unknown".to_string());
164
165                // white list
166                let is_my_code = symbol_name.starts_with("dsf_")
167                    || file_path.contains("/dsf-")
168                    || file_path.contains("DataSpringFlow");
169
170                // blacklist
171                let is_external_noise = file_path.contains(".cargo")
172                    || file_path.contains(".rustup")
173                    || symbol_name.starts_with("backtrace::")
174                    || symbol_name.contains("capture_backtrace")
175                    || symbol_name.contains("::Future::poll");
176
177                if is_my_code && !is_external_noise {
178                    let line = symbol.lineno().unwrap_or(0);
179                    frames.push(StackFrame {
180                        name: symbol_name,
181                        file: file_path,
182                        line,
183                    });
184                }
185            }
186        });
187
188        true // 继续向上追踪
189    });
190
191    if !frames.is_empty() {
192        frames.reverse();
193        let mut filtered_backtrace = String::new();
194        for (index, frame) in frames.iter().enumerate() {
195            filtered_backtrace.push_str(&format!(
196                "    {:>2}: {}\n            at {}:{}\n",
197                index, frame.name, frame.file, frame.line
198            ));
199        }
200
201        log::debug!(target: "backtrace","Filtered Backtrace (Python-style):\n{}", filtered_backtrace);
202    }
203}
